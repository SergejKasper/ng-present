ngShow ein neugeborenes HTML Element

		
angular.directive('ngShow', [function () {
	return {
		restrict: 'A',
		link: function (scope, element, attr) {
			scope.$watch(attr.ngShow, function ngShowWatchAction(value){
				element.css('display', toBoolean(value) ? '' : 'none');
			});
		}
	};
}]);
